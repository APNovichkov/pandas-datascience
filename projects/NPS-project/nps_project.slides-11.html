def get_NPS(n_pro, n_pass, n_det):
    return round((n_pro - n_det) / (n_pro + n_pass + n_det) * 100, 2)

print('NPS for 2017 data: {}'.format(get_NPS(num_promoters, num_passive, num_detractors)))
